#!/usr/bin/env python3
"""
wildlifesensor.py
-----------------
Raspberry Pi wildlife camera trap.

Hardware
--------
- PIR motion sensor on GPIO 4
- Raspberry Pi Camera Module (any revision)

Behaviour
---------
1. Wait for motion.
2. Start recording an H.264 video file with a timestamp-based name.
3. Keep recording for at least MIN_RECORD_SECONDS.
4. Keep recording until motion stops (or MAX_RECORD_SECONDS is reached).
5. Stop recording, log the clip, then go back to step 1.

All errors are caught so the sensor keeps running unattended.
"""

import logging
import os
import signal
import sys
import time
from datetime import datetime
from pathlib import Path

# ---------------------------------------------------------------------------
# Third-party imports – fail early with a helpful message if missing
# ---------------------------------------------------------------------------
try:
    from gpiozero import MotionSensor
except ImportError as exc:
    sys.exit(f"[FATAL] gpiozero not available: {exc}  (run: pip install gpiozero)")

try:
    from picamera import PiCamera
except ImportError as exc:
    sys.exit(f"[FATAL] picamera not available: {exc}  (run: pip install picamera)")

# ---------------------------------------------------------------------------
# Configuration – change these without touching the logic below
# ---------------------------------------------------------------------------
GPIO_PIN            = 4          # BCM pin number for the PIR DATA line
PIR_QUEUE_LEN       = 5          # samples the PIR averages before firing
CAMERA_RESOLUTION   = (1280, 720)
CAMERA_FRAMERATE    = 30
VIDEO_DIR           = Path("/home/pi/wildlife_videos")   # where clips are saved
MIN_RECORD_SECONDS  = 10.0       # always record at least this long
MAX_RECORD_SECONDS  = 300.0      # hard cap so a stuck PIR doesn't fill the disk
NO_MOTION_TIMEOUT   = 30.0       # give up waiting for motion-end after this many s
LOG_FILE            = VIDEO_DIR / "sensor.log"

# ---------------------------------------------------------------------------
# Logging
# ---------------------------------------------------------------------------
def _configure_logging() -> logging.Logger:
    VIDEO_DIR.mkdir(parents=True, exist_ok=True)
    fmt = "%(asctime)s  %(levelname)-8s  %(message)s"
    datefmt = "%Y-%m-%d %H:%M:%S"
    handlers: list[logging.Handler] = [
        logging.StreamHandler(sys.stdout),
        logging.FileHandler(LOG_FILE),
    ]
    logging.basicConfig(level=logging.INFO, format=fmt, datefmt=datefmt,
                        handlers=handlers)
    return logging.getLogger("wildlifesensor")

log = _configure_logging()

# ---------------------------------------------------------------------------
# Graceful shutdown on SIGINT / SIGTERM
# ---------------------------------------------------------------------------
_shutdown_requested = False

def _handle_signal(signum, _frame):
    global _shutdown_requested
    log.info("Shutdown signal %s received – finishing current clip then exiting.",
             signal.Signals(signum).name)
    _shutdown_requested = True

signal.signal(signal.SIGINT,  _handle_signal)
signal.signal(signal.SIGTERM, _handle_signal)

# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------
def _make_filename(directory: Path) -> Path:
    """Return a unique timestamped H.264 path inside *directory*."""
    now = datetime.now()
    name = now.strftime("%Y-%m-%d_%H-%M-%S") + ".h264"
    return directory / name


def _free_space_mb(path: Path) -> float:
    """Return free disk space in MB for the filesystem containing *path*."""
    stat = os.statvfs(path)
    return (stat.f_bavail * stat.f_frsize) / (1024 * 1024)


MIN_FREE_MB = 200  # refuse to start a clip if less than this is available


def _enough_disk_space() -> bool:
    free = _free_space_mb(VIDEO_DIR)
    if free < MIN_FREE_MB:
        log.warning("Low disk space: %.1f MB free (need %d MB) – skipping clip.",
                    free, MIN_FREE_MB)
        return False
    return True


# ---------------------------------------------------------------------------
# Camera context manager
# ---------------------------------------------------------------------------
def _open_camera() -> PiCamera:
    camera = PiCamera()
    camera.resolution  = CAMERA_RESOLUTION
    camera.framerate   = CAMERA_FRAMERATE
    # Brief warm-up so the sensor auto-exposure settles
    camera.start_preview()
    time.sleep(2)
    camera.stop_preview()
    log.info("Camera initialised at %dx%d @ %d fps.",
             CAMERA_RESOLUTION[0], CAMERA_RESOLUTION[1], CAMERA_FRAMERATE)
    return camera


# ---------------------------------------------------------------------------
# Core recording logic
# ---------------------------------------------------------------------------
def record_clip(camera: PiCamera) -> None:
    """Record one motion-triggered clip."""
    if not _enough_disk_space():
        return

    filepath = _make_filename(VIDEO_DIR)
    log.info("Motion detected – recording → %s", filepath.name)

    start_time = time.monotonic()
    recording  = False

    try:
        camera.start_recording(str(filepath))
        recording = True

        # ── Minimum buffer ──────────────────────────────────────────────────
        elapsed = 0.0
        while elapsed < MIN_RECORD_SECONDS:
            if _shutdown_requested:
                break
            time.sleep(0.1)
            elapsed = time.monotonic() - start_time

        # ── Wait for motion to stop (with hard cap) ──────────────────────
        if not _shutdown_requested:
            motion_end_deadline = time.monotonic() + NO_MOTION_TIMEOUT
            max_deadline        = start_time + MAX_RECORD_SECONDS

            while not _shutdown_requested:
                now_mono = time.monotonic()

                if now_mono >= max_deadline:
                    log.warning("Max recording duration (%ds) reached – stopping.",
                                MAX_RECORD_SECONDS)
                    break

                # pir.motion_detected reflects the current debounced state
                if not pir.motion_detected:
                    if now_mono >= motion_end_deadline:
                        break           # motion has been absent long enough
                else:
                    # Motion still ongoing – reset the no-motion deadline
                    motion_end_deadline = time.monotonic() + NO_MOTION_TIMEOUT

                time.sleep(0.1)

    except Exception as exc:
        log.error("Error during recording: %s", exc, exc_info=True)
    finally:
        if recording:
            try:
                camera.stop_recording()
            except Exception as exc:
                log.error("Error stopping recording: %s", exc, exc_info=True)

    duration = time.monotonic() - start_time
    file_mb  = filepath.stat().st_size / (1024 * 1024) if filepath.exists() else 0.0
    log.info("Clip saved: %s  (%.1f s, %.2f MB)", filepath.name, duration, file_mb)


# ---------------------------------------------------------------------------
# Main loop
# ---------------------------------------------------------------------------
def main() -> None:
    log.info("Wildlife sensor starting up.  Videos → %s", VIDEO_DIR)

    # Initialise PIR
    try:
        global pir
        pir = MotionSensor(GPIO_PIN, queue_len=PIR_QUEUE_LEN)
    except Exception as exc:
        log.critical("Could not initialise PIR on GPIO %d: %s", GPIO_PIN, exc,
                     exc_info=True)
        sys.exit(1)

    log.info("PIR calibrating on GPIO %d …", GPIO_PIN)
    # Allow the PIR sensor to settle before we start listening
    pir.wait_for_no_motion(timeout=30)
    log.info("PIR ready.  Waiting for motion …")

    camera: PiCamera | None = None
    try:
        camera = _open_camera()

        while not _shutdown_requested:
            # wait_for_motion accepts a timeout so we can check _shutdown_requested
            detected = pir.wait_for_motion(timeout=5)
            if detected is None:
                # Timed out – no motion; loop again to check shutdown flag
                continue

            record_clip(camera)

            if not _shutdown_requested:
                log.info("Motion ended – waiting for next event …")

    except Exception as exc:
        log.critical("Unhandled exception in main loop: %s", exc, exc_info=True)
        sys.exit(1)
    finally:
        if camera is not None:
            try:
                camera.close()
                log.info("Camera closed.")
            except Exception:
                pass
        log.info("Wildlife sensor shut down cleanly.")


if __name__ == "__main__":
    main()
